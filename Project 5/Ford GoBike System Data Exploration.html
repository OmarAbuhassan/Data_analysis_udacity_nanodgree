# Mask to delete outliers
from scipy import stats
def delete_outliers(df):
    if isinstance(df, pd.Series):
        df = df.to_frame()
    z = np.abs(stats.zscore(df))
    delete_outliers = (z < 3).all(axis=1)
    return delete_outliers
